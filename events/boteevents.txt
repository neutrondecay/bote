character_event = { #Independent power embraces Renaissance
	id = bote.00005
	only_playable = yes
	only_independent = yes
	has_global_flag = renaissance_started
	hide_window = yes
	
	immediate = { #Get 25% of total number of vassals
		set_variable = {
			which = local_count
			value = 1
		}
		any_realm_character = {
				limit = {
					is_playable = yes
				}
				change_variable = {
					which = local_count
					value = 1
				}
		}
		divide_variable = {
			which = local_count
			value = 4
		}
		log = "[local_count.GetValue]"
	}
	trigger = {
		NOT = {
			primary_title = {
				has_title_flag = renaissance_realm
			}
		}
		any_realm_lord = {
			has_character_flag = renaissance_ruler
			count >= local_count
		}
	}
	
	mean_time_to_happen = {
		months = 24
	}
	option = {
		any_realm_lord = {
			limit = {
				ai = no
			}
			narrative_event = {
				id = bote.00006
			}
		}
		narrative_event = {
			id = bote.00006
		}
		primary_title = {
			set_title_flag = renaissance_realm
		}
	}
}
